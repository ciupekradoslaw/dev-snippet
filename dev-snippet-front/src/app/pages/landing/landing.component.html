// JWT Token Validation
export async function validateToken(
  token: string
): Promise<JWTPayload> {
  try {
    const decoded = await verify(
      token, secret, 'HS256'
    );
    return decoded;
  } catch {
    throw new UnauthorizedException();
  }
}
